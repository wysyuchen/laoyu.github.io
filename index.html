import requests from bs4 import BeautifulSoup import json def scrape_gaoxiaorencai(): url = "https://www.gaoxiaojob.com/zhaopin/" headers = { "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/96.0.4664.45 Safari/537.36" } response = requests.get(url, headers=headers) if response.status_code == 200: soup = BeautifulSoup(response.content, 'html.parser') jobs = [] # 假设每条招聘信息在某个 div 或 li 标签中 for job in soup.select('.joblist > li'): # 需要根据网站实际结构调整选择器 title = job.find('a').text.strip() link = job.find('a')['href'] date = job.find('span', class_='date').text.strip() # 也根据实际情况调整 jobs.append({ 'title': title, 'link': link, 'date': date }) return jobs else: print("Failed to retrieve data") return [] if __name__ == "__main__": job_listings = scrape_gaoxiaorencai() with open("jobs.json", "w", encoding="utf-8") as f: json.dump(job_listings, f, ensure_ascii=False, indent=4)
